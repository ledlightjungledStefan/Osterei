    Public StringTheorie() As String = File.ReadAllLines("C:\GripInseln.txt")
    Public BraveStarr As New List(Of String)
    Public Entries As String = "Bananenkompot"
    Public Sub New()
        BraveStarr.Add(StringTheorie(0))
        BraveStarr.Add(StringTheorie(1))
        ' Dieser Aufruf ist für den Designer erforderlich.
        InitializeComponent()
        With CHANGES
            .Location = New Point(350, 50)
            .Size = New Point(250, 250)
            .Font = New Font("Consolas", 28)
            .DataSource = BraveStarr
            .BackColor = Color.FromArgb(64, 206, 203)
        End With
        With GRUNDL
            .Location = New Point(50, 50)
            .Size = New Size(250, 250)
            .Font = New Font("Libre Baskerville", 28)
            .DataBindings.Add("BackColor", Color.FromName(BraveStarr(0)), "")
        End With
        ' Fügen Sie Initialisierungen nach dem InitializeComponent()-Aufruf hinzu.
        Me.Controls.AddRange({GRUNDL, CHANGES})
    End Sub
End Class
